service cloud.firestore {
  match /databases/{database}/documents {

    match /sharedBoxes/{boxId} {
      allow read: if true;

      allow create: if request.resource.data.name is string
                    && request.resource.data.name.size() > 0
                    && request.resource.data.name.size() <= 100
                    && request.resource.data.items is list
                    && request.resource.data.items.size() > 0
                    && request.resource.data.items.size() <= 50;

      allow update: if request.resource.data.diff(resource.data).affectedKeys()
                      .hasOnly(['pullHistory', 'updatedAt', 'lastActivity']);

      allow delete: if resource.data.creatorDeviceId is string
                    && resource.data.creatorDeviceId != ''
                    && request.resource == null;
    }

    match /boxCatalog/{boxId} {
      allow read: if true;
      allow write: if false;
    }

    match /purchases/{purchaseId} {
      allow read, write: if false;
    }

    match /users/{userId} {
      allow read, write: if false;
    }

    match /boxTemplates/{templateId} {
      allow read: if true;

      allow create: if request.resource.data.name is string
                    && request.resource.data.name.size() > 0
                    && request.resource.data.name.size() <= 100
                    && request.resource.data.items is list
                    && request.resource.data.items.size() > 0
                    && request.resource.data.items.size() <= 50;

      allow update, delete: if false;
    }
  }
}

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /sharedBoxes/{boxId} {
      allow read: if true;

      allow create: if request.resource.data.name is string
                    && request.resource.data.name.size() > 0
                    && request.resource.data.name.size() <= 100
                    && request.resource.data.items is list
                    && request.resource.data.items.size() > 0
                    && request.resource.data.items.size() <= 50;

      // Allow pull history updates from anyone (participants opening boxes)
      // Allow full edits from the box creator (matched by creatorDeviceId)
      allow update: if request.resource.data.diff(resource.data).affectedKeys()
                      .hasOnly(['pullHistory', 'updatedAt', 'lastActivity'])
                    || resource.data.creatorDeviceId == request.resource.data.creatorDeviceId;

      allow delete: if true;
    }

    match /boxCatalog/{boxId} {
      allow read: if true;
      allow write: if false;
    }

    match /purchases/{purchaseId} {
      allow read, write: if false;
    }

    match /users/{userId} {
      allow read, write: if false;
    }

    match /boxTemplates/{templateId} {
      allow read: if true;

      allow create: if request.resource.data.name is string
                    && request.resource.data.name.size() > 0
                    && request.resource.data.name.size() <= 100
                    && request.resource.data.items is list
                    && request.resource.data.items.size() > 0
                    && request.resource.data.items.size() <= 50;

      allow update, delete: if false;
    }
  }
}
