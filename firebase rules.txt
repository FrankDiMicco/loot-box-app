    match /sharedBoxes/{boxId} {
      allow read: if true;
      allow write: if true;
    }
    match /boxCatalog/{boxId} {
      allow read: if true;
      allow write: if true;
    }
    match /purchases/{purchaseId} {
      allow read, write: if true;
    }

      allow read, write: if true;
    }
  }
}
